// Definição do tamanho máximo dos vetores
const MAX_CLIENTES = 100
const MAX_DESTINOS = 50

// Declaração dos vetores para clientes e destinos
vetor clientes[MAX_CLIENTES]
vetor destinos[MAX_DESTINOS]

// Variáveis auxiliares

inteiro numClientes = 0
inteiro numDestinos = 0

função menuPrincipal():
    repetir
        escrever("-------- MENU PRINCIPAL --------")
        escrever("1. Cadastro de cliente")
        escrever("2. Cadastro de destino")
        escrever("3. Consulta de clientes cadastrados")
        escrever("4. Consulta de destinos disponíveis")
        escrever("5. Sair")
        escrever("Escolha uma opção: ")
        ler opcao

        escolher opcao
            caso 1:
                cadastrarCliente()
            caso 2:
                cadastrarDestino()
            caso 3:
                consultaClientes()
            caso 4:
                consultaDestinos()
            caso 5:
                escrever("Encerrando o programa...")
                parar
            caso contrário:
                escrever("Opção inválida! Tente novamente.")

    enquanto verdadeiro

fim função

função cadastrarCliente():
    se numClientes < MAX_CLIENTES então
        escrever("-------- CADASTRO DE CLIENTE --------")
        escrever("Nome do cliente: ")
        ler nomeCliente

        // Outros dados do cliente podem ser solicitados aqui, como CPF, telefone, etc.

        clientes[numClientes] = nomeCliente
        numClientes = numClientes + 1
        escrever("Cliente cadastrado com sucesso!")
    senão
        escrever("Limite máximo de clientes cadastrados atingido!")
    fim se

fim função

função cadastrarDestino():
    se numDestinos < MAX_DESTINOS então
        escrever("-------- CADASTRO DE DESTINO --------")
        escrever("Nome do destino: ")
        ler nomeDestino

        // Outros dados do destino podem ser solicitados aqui, como preço, descrição, etc.

        destinos[numDestinos] = nomeDestino
        numDestinos = numDestinos + 1
        escrever("Destino cadastrado com sucesso!")
    senão
        escrever("Limite máximo de destinos cadastrados atingido!")
    fim se

fim função

função consultaClientes():
    escrever("-------- CLIENTES CADASTRADOS --------")
    para i de 0 até numClientes - 1 passo 1 faça
        escrever("Cliente ", i + 1, ": ", clientes[i])
    fim para

fim função

função consultaDestinos():
    escrever("-------- DESTINOS DISPONÍVEIS --------")
    para i de 0 até numDestinos - 1 passo 1 faça
        escrever("Destino ", i + 1, ": ", destinos[i])
    fim para

fim função

// Programa principal
menuPrincipal()
